    consensus_units = []
    for sg in zip(sg_units, sg_sorters):

    curated_not_split = set(curated.get_unit_ids()).intersection(sortings[1].get_unit_ids())

    not_curated = set(consensus_units).difference(curated_not_split)
    not_consensus = set(curated_not_split).difference(consensus_units)
